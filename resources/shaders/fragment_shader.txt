#version 330 core
out vec4 FragColor;

in vec3 color;
in vec2 tex_coord;

uniform sampler2D tex0;
uniform sampler2D tex1;
uniform bool is_tex_set;

void main()
{
    vec4 temp_frag_color = vec4(color, 1.0f);
    if (is_tex_set)
    {
        temp_frag_color = mix(texture(tex0, tex_coord), texture(tex1, tex_coord), 0.5f);
    }
    FragColor = temp_frag_color;
}
